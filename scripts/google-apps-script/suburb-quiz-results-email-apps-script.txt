function sendPendingSuburbQuizEmails() {
  const ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);
  const sheet = getOrCreateSheet_(ss, CONFIG.SHEET_NAME);
  ensureQuizEmailColumns_(sheet);

  const indexes = headerIndexes_(sheet);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) {
    return { ok: true, scanned: 0, matched: 0, sent: 0, failed: 0 };
  }

  const values = sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).getValues();
  let matched = 0;
  let sent = 0;
  let failed = 0;

  values.forEach(function (row, idx) {
    const rowNumber = idx + 2;
    const sourcePage = clean_(row[indexes.source_page]);
    const sourceKey = clean_(row[indexes.source_key]);
    const email = clean_(row[indexes.email]);
    const status = clean_(row[indexes.quiz_email_status]).toLowerCase();

    const sourceHaystack = (sourcePage + ' ' + sourceKey).toLowerCase();
    const isQuizSource = sourceHaystack.indexOf(CONFIG.SOURCE_MATCH) !== -1 || sourceHaystack.indexOf('raleigh-suburb-quiz') !== -1;
    if (!isQuizSource || !email || status === 'sent') {
      return;
    }

    matched += 1;
    try {
      const payload = rowToPayload_(row, indexes);
      sendSuburbResultsEmail_(payload);
      markQuizEmailStatus_(sheet, rowNumber, 'sent', new Date(), '');
      sent += 1;
    } catch (err) {
      const message = clean_(err && err.message ? err.message : err);
      markQuizEmailStatus_(sheet, rowNumber, 'failed', '', message);
      failed += 1;
    }
  });

  return { ok: true, scanned: values.length, matched: matched, sent: sent, failed: failed };
}

function installSuburbQuizEmailTrigger() {
  removeSuburbQuizEmailTriggers();
  ScriptApp.newTrigger('sendPendingSuburbQuizEmails')
    .timeBased()
    .everyMinutes(10)
    .create();
  return { ok: true, message: 'Trigger installed: sendPendingSuburbQuizEmails every 10 minutes.' };
}

function removeSuburbQuizEmailTriggers() {
  const triggers = ScriptApp.getProjectTriggers();
  let removed = 0;
  triggers.forEach(function (trigger) {
    if (trigger.getHandlerFunction() === 'sendPendingSuburbQuizEmails') {
      ScriptApp.deleteTrigger(trigger);
      removed += 1;
    }
  });
  return { ok: true, removed: removed };
}

function runSuburbQuizEmailNow() {
  return sendPendingSuburbQuizEmails();
}

function sendSuburbResultsEmail_(lead, isTest) {
  const toEmail = clean_(lead.email);
  if (!toEmail) throw new Error('Missing lead email.');

  const topPick = clean_(lead.quiz_top_pick);
  const secondPick = clean_(lead.quiz_second_pick);
  const thirdPick = clean_(lead.quiz_third_pick);

  if (!topPick) throw new Error('Missing quiz top pick.');

  const picks = [topPick, secondPick, thirdPick].filter(function (v, i, arr) {
    return v && arr.indexOf(v) === i;
  });

  const suburbCards = picks.map(function (suburbName, idx) {
    return buildSuburbBlockHtml_(suburbName, idx === 0);
  }).join('');

  const firstName = clean_(lead.first_name) || 'there';
  const subject = CONFIG.SUBJECT_LINE;

  const htmlBody =
    '<div style="font-family:Arial,sans-serif;color:#1F2937;line-height:1.55;max-width:760px;margin:0 auto;padding:20px;">' +
      '<div style="text-align:center;margin:0 0 6px;"><img src="' + escapeHtml_(CONFIG.LOGO_URL) + '" alt="Living in Raleigh Team" style="max-width:280px;width:100%;height:auto;"></div>' +
      '<h1 style="text-align:center;color:#2F5233;margin:0 0 8px;">Your Raleigh Suburb Quiz Results</h1>' +
      '<p style="text-align:center;margin:0 0 24px;">Hi ' + escapeHtml_(firstName) + ', here are your personalized suburb matches based on your quiz responses.</p>' +
      suburbCards +
      '<div style="margin-top:24px;padding:16px;border:1px solid #d9e2e6;border-radius:12px;background:#f7fbf8;">' +
        '<h3 style="margin:0 0 8px;color:#2F5233;">Want help choosing the best fit?</h3>' +
        '<p style="margin:0 0 10px;">We can help you compare commute, schools, home styles, and budget in real time.</p>' +
        '<a href="' + CONFIG.BASE_SITE_URL + '/consult/" style="display:inline-block;background:#2F5233;color:#fff;text-decoration:none;padding:10px 14px;border-radius:999px;font-weight:700;">Book Your Suburb Strategy Call</a>' +

  if (!suburb) {
    return '<div style="margin:18px 0;"><h2 style="text-align:center;">' + escapeHtml_(suburbName) + '</h2></div>';
  }

  const badge = isTopPick
    ? '<div style="text-align:center;margin-bottom:8px;"><span style="background:#2F5233;color:#fff;padding:6px 10px;border-radius:999px;font-size:12px;font-weight:700;">Top Pick</span></div>'

  return (
    '<section style="margin:22px 0;padding:18px;border:1px solid #dce5ea;border-radius:14px;background:#fff;">' +
      badge +
      '<h2 style="text-align:center;margin:0 0 12px;color:#1F2937;font-size:28px;font-weight:800;">' + escapeHtml_(suburbName) + '</h2>' +
      (suburb.imageUrl ?       '<p style="margin:8px 0 0;"><a href="' + escapeHtml_(getSuburbLearnMoreUrl_(suburbName)) + '" target="_blank" rel="noopener noreferrer" style="color:#2F5233;font-weight:700;">Learn More About ' + escapeHtml_(suburbName) + '</a></p>' +

function rowToPayload_(row, indexes) {
  return {
    first_name: clean_(row[indexes.first_name]),
    last_name: clean_(row[indexes.last_name]),
    email: clean_(row[indexes.email]),
    source_page: clean_(row[indexes.source_page]),
    source_key: clean_(row[indexes.source_key]),
    quiz_top_pick: clean_(row[indexes.quiz_top_pick]),
    quiz_second_pick: clean_(row[indexes.quiz_second_pick]),
    quiz_third_pick: clean_(row[indexes.quiz_third_pick])
  };
}

function ensureQuizEmailColumns_(sheet) {
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(function (h) { return clean_(h); });
  [CONFIG.STATUS_COLUMN, CONFIG.SENT_AT_COLUMN, CONFIG.ERROR_COLUMN].forEach(function (name) {
    if (headers.indexOf(name) === -1) {
      sheet.insertColumnAfter(sheet.getLastColumn());
      sheet.getRange(1, sheet.getLastColumn()).setValue(name);
    }
  });
}

function markQuizEmailStatus_(sheet, rowNumber, status, sentAt, errorMessage) {
  const indexes = headerIndexes_(sheet);
  const statusCol = indexes.quiz_email_status + 1;
  const sentCol = indexes.quiz_email_sent_at + 1;
  const errorCol = indexes.quiz_email_error + 1;
  if (statusCol < 1 || sentCol < 1 || errorCol < 1) return;

  sheet.getRange(rowNumber, statusCol).setValue(clean_(status));
  sheet.getRange(rowNumber, sentCol).setValue(sentAt || '');
  sheet.getRange(rowNumber, errorCol).setValue(clean_(errorMessage));
}

function getOrCreateSheet_(ss, name) {
  let sh = ss.getSheetByName(name);
  if (!sh) sh = ss.insertSheet(name);
  return sh;
}

function headerIndexes_(sheet) {
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const idx = function (name) { return headers.indexOf(name); };
  return {
    first_name: idx('First Name'),
    last_name: idx('Last Name'),
    email: idx('Email'),
    source_page: idx('Source Page'),
    source_key: idx('Source Key'),
    quiz_top_pick: idx('Quiz Top Pick'),
    quiz_second_pick: idx('Quiz Second Pick'),
    quiz_third_pick: idx('Quiz Third Pick'),
    quiz_email_status: idx(CONFIG.STATUS_COLUMN),
    quiz_email_sent_at: idx(CONFIG.SENT_AT_COLUMN),
    quiz_email_error: idx(CONFIG.ERROR_COLUMN)
  };
}

function clean_(v) {
  return String(v == null ? '' : v).replace(/\s+/g, ' ').trim();
}

function escapeHtml_(value) {
  return String(value == null ? '' : value)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

function shuffle_(arr) {
  const out = arr.slice();
  for (let i = out.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    const tmp = out[i];
    out[i] = out[j];
    out[j] = tmp;
  }
  return out;
}
